
void main (){
    while (true){
        if (зубная паста не выдавливается из тюбика){
            switch (степень наполненности тюбика) {
                case паста почти заканчивается:
                    сгрести пасту к горлышку();
                case паста заканчивается:
                    положить тюбик с зубной пастой на твердую поверхность();
                    пластиковой карточкой сгрести пасту к горлышку();
                case пасты почти нет:
                    купить зубную пасту();
                    разрезать тюбик();
                case тюбик разрезан:
                    снять с тюбика зажим();
                case пасты в тюбике нет:
                    выбросить тюбик();
            }
        }
    }
    почистить зубы();
    if (тюбик разрезан) {
        закрыть зажимом тюбик();
    } else {
        continue;
    }
}
